camelize.js

/**
 * @description 下划线转驼峰
 * @param str
 * @returns {*|void|string}
 */
function camelize(str) {
    let camelizeRe = /-(\w)/g;
    return str.replace(camelizeRe, function(_, c) {
        return c ? c.toUpperCase() : '';
    })
}


export default camelize
